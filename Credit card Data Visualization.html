# You need to find top 10 customers for each city in terms of their repayment amount by 
#different products and by different time periods i.e. year or month. The user should be able 
#to specify the product (Gold/Silver/Platinum) and time period (yearly or monthly) and the 
#function should automatically take these inputs while identifying the top 10 customers.
def f_prodct_year (product, timeperiod):
    if product.lower() =='gold' and timeperiod =='monthly':
        result = final.loc[final.Product =='Gold'].groupby([(final.Month_y.dt.month),'City','Customer']).Amount_y.sum().reset_index()
        f_res = result.sort_values(['Month_y','City','Amount_y'],ascending = [True,True,False])
    elif product.lower() =='platinum' and timeperiod =='monthly':
        result = final.loc[final.Product =='Platimum'].groupby([(final.Month_y.dt.month),'City','Customer']).Amount_y.sum().reset_index()
        f_res = result.sort_values(['Month_y','City','Amount_y'],ascending = [True,True,False])
    if product.lower() =='silver' and timeperiod=='monthly':
        result = final.loc[final.Product =='Silver'].groupby([(final.Month_y.dt.month),'City','Customer']).Amount_y.sum().reset_index()
        f_res = result.sort_values(['Month_y','City','Amount_y'],ascending = [True,True,False])
    elif product.lower() =='gold' and timeperiod =='yearly':
        result = final.loc[final.Product =='Gold'].groupby([(final.Month_y.dt.year),'City','Customer']).Amount_y.sum().reset_index()
        f_res = result.sort_values(['Month_y','City','Amount_y'],ascending = [True,True,False])
    if product.lower() =='silver' and timeperiod =='yearly':
        result =final.loc[final.Product =='Silver'].groupby([(final.Month_y.dt.year),'City','Customer']).Amount_y.sum().reset_index()
        f_res = result.sort_values(['Month_y','City','Amount_y'],ascending = [True,True,False])
    elif product.lower() =='platinum' and timeperiod =='yearly':
        result = final.loc[final.Product =='Platimum'].groupby([(final.Month_y.dt.year),'City','Customer']).Amount_y.sum().reset_index()
        f_res = result.sort_values(['Month_y','City','Amount_y'],ascending = [True,True,False])
    return f_res
